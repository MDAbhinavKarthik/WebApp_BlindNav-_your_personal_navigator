# -------------------- Multilingual TTS --------------------
class MultilingualTTS:
    """Offline-first TTS using pyttsx3 when available. Falls back to print()."""
    
    _instance_lock = threading.Lock()
    _instance = None
    _initialized = False
    
    def __new__(cls, lang="en"):
        with cls._instance_lock:
            if cls._instance is None:
                cls._instance = super(MultilingualTTS, cls).__new__(cls)
            return cls._instance
    
    def __init__(self, lang="en"):
        if self._initialized:
            return
            
        with self._instance_lock:
            if not self._initialized:
                self.lang = lang
                self._use_engine = TTS_AVAILABLE
                self.queue = queue.Queue()
                self._stop_flag = threading.Event()
                self.rate = 175  # Default speaking rate (can be adjusted)
                self.volume = 0.9  # Default volume (0.0 to 1.0)
                
                if self._use_engine:
                    try:
                        self.engine = pyttsx3.init()
                        self._set_voice(lang)
                        self._configure_engine()
                        # background thread processes queued non-blocking speech
                        self.thread = threading.Thread(target=self._loop, daemon=True)
                        self.thread.start()
                        # Test the engine silently (no speech during initialization)
                        # self.speak_blocking("Screen reader initialized.", fallback_print=False)
                        self._initialized = True
                        print("[TTS] Text-to-speech engine initialized successfully")
                        # Test: speak success line after full init
                        self.engine.say("Speech synthesis is working.")
                        self.engine.runAndWait()
                    except Exception as e:
                        self._use_engine = False
                        print(f"[TTS] pyttsx3 failed to initialize: {e} -> falling back to print()")
                        print("[TTS] To enable voice, please ensure pyttsx3 is installed: pip install pyttsx3")
                        try:
                            import platform
                            if platform.system() == "Windows":
                                import winsound
                                winsound.MessageBeep()
                        except Exception:
                            pass
                    print("[TTS] To enable voice, please ensure pyttsx3 is installed: pip install pyttsx3")
                else:
                    print("[TTS] pyttsx3 not available -> using print fallback")
                    print("[TTS] To enable voice, please install pyttsx3: pip install pyttsx3")
                    try:
                        import platform
                        if platform.system() == "Windows":
                            import winsound
                            winsound.MessageBeep()
                    except Exception:
                        pass

    def _configure_engine(self):
        """Configure TTS engine properties."""
        if not self._use_engine:
            return
        try:
            self.engine.setProperty('rate', self.rate)
            self.engine.setProperty('volume', self.volume)
        except Exception as e:
            print(f"[TTS] Warning: Could not configure engine properties: {e}")

    def _set_voice(self, lang):
        """Set the voice based on language preference with improved matching."""
        if not self._use_engine:
            return

        try:
            voices = self.engine.getProperty('voices')
            if not voices:
                print("[TTS] No voices available")
                return "default voice"

            # Define language mappings and alternatives
            lang_mappings = {
                "hi": ["hindi", "hi", "hi-in", "ind"],
                "kn": ["kannada", "kn", "kn-in", "kan"],
                "en": ["english", "en", "en-us", "en-gb", "eng"]
            }

            # Get possible matches for the requested language
            target_patterns = lang_mappings.get(lang.lower(), [lang.lower()])
            
            # Score each voice for suitability
            voice_scores = []
            for v in voices:
                score = 0
                name_lower = v.name.lower()
                
                # Check voice name and properties
                for pattern in target_patterns:
                    if pattern in name_lower:
                        score += 2
                    if hasattr(v, 'languages') and any(pattern in lang.lower() for lang in v.languages):
                        score += 2
                
                # Extra points for matching language code
                if hasattr(v, 'id') and lang.lower() in v.id.lower():
                    score += 1
                
                # Prefer Microsoft voices over SAPI voices for better quality
                if "microsoft" in name_lower:
                    score += 1
                
                voice_scores.append((v, score))
            
            # Sort by score and get the best match
            voice_scores.sort(key=lambda x: x[1], reverse=True)
            best_match = voice_scores[0][0] if voice_scores else voices[0]
            
            # Configure voice properties
            self.engine.setProperty('voice', best_match.id)
            
            # Adjust rate and properties based on language
            if lang != "en":
                # Slower rate for non-English languages for better clarity
                self.engine.setProperty('rate', self.rate * 0.9)
            else:
                self.engine.setProperty('rate', self.rate)
            
            name = best_match.name
            print(f"[TTS] Selected voice: {name} for language: {lang}")
            # print(f"[TTS] Voice properties: Rate={self.engine.getProperty('rate')}, Volume={self.engine.getProperty('volume')}")
            
            return name
            
        except Exception as e:
            print(f"[TTS] Warning: Could not set voice: {e}")
            print("[TTS] Falling back to default voice")
            return "default voice"

    def set_lang(self, lang):
        """Change the language and voice with proper synchronization."""
        if not self._use_engine:
            print(f"[TTS] Language changed to {lang} (text-only mode)")
            return

        prev_lang = self.lang
        self.lang = lang
        
        # Ensure queue is empty before changing voice
        while not self.queue.empty():
            try:
                self.queue.get_nowait()
                self.queue.task_done()
            except Exception:
                pass
                
        try:
            voice_name = self._set_voice(lang)
            
            # Prepare confirmation message
            confirmations = {
                "en": "English selected",
                "hi": "हिंदी चुनी गई",
                "kn": "ಕನ್ನಡ ಆಯ್ಕೆಯಾಗಿದೆ"
            }
            
            # Wait for previous speech to finish
            time.sleep(0.5)
            
            # Speak confirmation in the new language
            self.speak_blocking(confirmations.get(lang, confirmations["en"]), 
                             fallback_print=True)
            
            return True
            
        except Exception as e:
            print(f"[TTS] Error during language switch: {e}")
            self.speak_blocking("Error changing language", fallback_print=True)
            return False
        
    def set_rate(self, rate):
        """Adjust speaking rate (words per minute)."""
        self.rate = max(50, min(400, rate))  # Clamp between 50-400 WPM
        if self._use_engine:
            try:
                self.engine.setProperty('rate', self.rate)
                self.speak(f"Speaking rate set to {self.rate} words per minute")
            except Exception as e:
                print(f"[TTS] Could not set rate: {e}")

    def set_volume(self, volume):
        """Adjust speaking volume (0.0 to 1.0)."""
        self.volume = max(0.0, min(1.0, volume))
        if self._use_engine:
            try:
                self.engine.setProperty('volume', self.volume)
                self.speak(f"Volume set to {int(self.volume * 100)} percent")
            except Exception as e:
                print(f"[TTS] Could not set volume: {e}")

    def _loop(self):
        while not self._stop_flag.is_set():
            try:
                text = self.queue.get(timeout=0.5)
            except Exception:
                continue
            if text is None:
                break
            try:
                if self._use_engine:
                    # Check if engine is already running
                    if hasattr(self.engine, '_inLoop') and self.engine._inLoop:
                        # Engine is busy, skip this text
                        print(f"[TTS] Engine busy, skipping: {text[:50]}...")
                        continue
                    
                    self.engine.say(text)
                    self.engine.runAndWait()
                else:
                    print("[TTS]", text)
            except Exception as e:
                print(f"[TTS] error: {e}")
                # If engine is in a bad state, try to reinitialize
                if "run loop already started" in str(e):
                    try:
                        self.engine.stop()
                        time.sleep(0.1)
                    except:
                        pass
            finally:
                try:
                    self.queue.task_done()
                except Exception:
                    pass

    def speak(self, text, message_type="info"):
        """
        Non-blocking speak — queued.
        message_type can be: "info", "warning", "error", "success"
        """
        if not text:
            return

        # Format message based on type
        prefix = ""
        if message_type == "warning":
            prefix = "Warning: "
        elif message_type == "error":
            prefix = "Error: "
        elif message_type == "success":
            prefix = "Success: "
        
        full_text = prefix + text
        
        # Dim background music before speaking
        if hasattr(self, 'audio_manager') and hasattr(self.audio_manager, '_music_player'):
            self.audio_manager._music_player.dim()
        
        if self._use_engine:
            self.queue.put(full_text)
        else:
            tag = f"[TTS/{message_type.upper()}]"
            print(f"{tag} {text}")
            
        # Schedule music volume restoration after speech
        if hasattr(self, 'audio_manager') and hasattr(self.audio_manager, '_music_player'):
            threading.Timer(len(text) * 0.1 + 0.5, self.audio_manager._music_player.restore).start()
            
    def speak_blocking(self, text, message_type="info", fallback_print=True, urgent=False):
        """
        Ensure text is spoken (blocking). Useful before listen() calls.
        message_type can be: "info", "warning", "error", "success"
        urgent=True will increase volume temporarily for important messages
        """
        if not text:
            return
            
        # Format message based on type
        prefix = ""
        if message_type == "warning":
            prefix = "Warning: "
        elif message_type == "error":
            prefix = "Error: "
        elif message_type == "success":
            prefix = "Success: "
            
        full_text = prefix + text
        
        # Dim background music before speaking
        if hasattr(self, 'audio_manager') and hasattr(self.audio_manager, '_music_player'):
            self.audio_manager._music_player.dim()
        
        if self._use_engine:
            try:
                # Temporarily boost volume for urgent messages
                original_volume = None
                if urgent and self._use_engine:
                    try:
                        original_volume = self.engine.getProperty('volume')
                        self.engine.setProperty('volume', min(1.0, original_volume * 1.5))
                    except Exception:
                        pass
                
                # Use queue for blocking speech to avoid runAndWait conflicts
                self.queue.put(full_text)
                
                # Wait for the speech to complete
                import time
                start_time = time.time()
                while not self.queue.empty() and (time.time() - start_time) < 10:  # 10 second timeout
                    time.sleep(0.1)
                
                # Restore original volume
                if original_volume is not None:
                    try:
                        self.engine.setProperty('volume', original_volume)
                    except Exception:
                        pass
                        
            except Exception as e:
                print(f"[TTS] Engine error: {e}")
                if fallback_print:
                    tag = f"[TTS/{message_type.upper()}]"
                    print(f"{tag} {text}")
        else:
            if fallback_print:
                tag = f"[TTS/{message_type.upper()}]"
                print(f"{tag} {text}")
                
        # Restore background music volume after speaking
        if hasattr(self, 'audio_manager') and hasattr(self.audio_manager, '_music_player'):
            self.audio_manager._music_player.restore()

    def stop(self):
        if self._use_engine:
            self._stop_flag.set()
            self.queue.put(None)
            if hasattr(self, '_engine'):
                try:
                    self._engine.stop()
                except Exception:
                    pass
            # Clear the queue
            while not self.queue.empty():
                try:
                    self.queue.get_nowait()
                except queue.Empty:
                    break

    def speak_direct_blocking(self, text):
        if self._use_engine:
            self.engine.say(text)
            self.engine.runAndWait()
        else:
            print(text)

class VoiceIO:
    def __init__(self, lang="en", audio_manager=None):
        self.lang = lang
        self.recognizer = None
        self.mic_available = False
        self.audio_manager = audio_manager
        if SR_AVAILABLE:
            try:
                self.recognizer = sr.Recognizer()
                # Try to find a suitable microphone, preferring Realtek arrays
                mics = sr.Microphone.list_microphone_names()
                mic_index = None
                for i, name in enumerate(mics):
                    if "Microphone Array" in name:
                        mic_index = i
                        # Using microphone - voice-only interface
                        break
                
                if mic_index is not None:
                    self.microphone = sr.Microphone(device_index=mic_index)
                else:
                    self.microphone = sr.Microphone()
                    # Using default microphone - voice-only interface
                
                # Test the microphone
                with self.microphone as source:
                    self.recognizer.adjust_for_ambient_noise(source, duration=1)
                self.mic_available = True
                # Microphone initialized successfully - voice-only interface
            except Exception as e:
                # microphone not accessible; fallback to text input
                self.microphone = None
                self.mic_available = False
                # Microphone setup failed - voice-only interface
                # Falling back to typed input
        else:
            # speech_recognition not available, falling back to typed input - voice-only interface
            pass

    def set_lang(self, lang):
        """Set the recognition language with appropriate variants."""
        self.lang = lang
        # Reset recognizer to clear any cached settings
        if self.recognizer:
            self.recognizer = sr.Recognizer()
            # Adjust recognition parameters for different languages
            if lang != "en":
                # More lenient recognition for non-English
                self.recognizer.operation_timeout = 30  # Increased timeout
                self.recognizer.pause_threshold = 1.0   # More time between phrases
            else:
                # Default settings for English
                self.recognizer.operation_timeout = 20
                self.recognizer.pause_threshold = 0.8

    def listen(self, timeout=7, phrase_time_limit=None, prompt=None, max_retries=2):
        """
        Try to listen via microphone. If not available or recognition fails,
        fallback to typed input for debugging/testing.
        Returns lowercase string or None.
        """
        if not (self.mic_available and self.recognizer and self.microphone):
            return self._typed_input_fallback(prompt)
            
        if prompt:
            print("[Listening prompt]:", prompt)
        
        # Dim background music before listening
        if hasattr(self, 'audio_manager'):
            self.audio_manager.dim_background_music()
        
        retries = 0
        try:
            while retries <= max_retries:
                try:
                    with self.microphone as source:
                        # Enhanced ambient noise adjustment for multilingual support
                        if self.lang == "kn":  # Kannada needs more adjustment
                            noise_duration = 2.0
                            self.recognizer.energy_threshold = 5000  # More sensitive for Kannada
                        elif self.lang == "hi":  # Hindi needs moderate adjustment
                            noise_duration = 1.8
                            self.recognizer.energy_threshold = 4500  # Moderate sensitivity for Hindi
                        else:  # English
                            noise_duration = 0.5
                            self.recognizer.energy_threshold = 3000
                        
                        self.recognizer.adjust_for_ambient_noise(source, duration=noise_duration)
                        # Additional settings for better multilingual recognition
                        self.recognizer.dynamic_energy_threshold = True
                        self.recognizer.pause_threshold = 1.0 if self.lang != "en" else 0.8
                        # Voice-only interface - no console output
                        
                        try:
                            # Enhanced timeout for multilingual support
                            if self.lang == "kn":  # Kannada needs more time
                                actual_timeout = timeout * 2.0
                            elif self.lang == "hi":  # Hindi needs moderate extra time
                                actual_timeout = timeout * 1.8
                            else:  # English
                                actual_timeout = timeout
                            audio = self.recognizer.listen(source, timeout=actual_timeout, 
                                                         phrase_time_limit=phrase_time_limit)
                        except sr.WaitTimeoutError:
                            # No speech detected - voice-only interface
                            retries += 1
                            continue
                        
                        try:
                            # Enhanced multilingual recognition with multiple engines
                            text = None
                            errors = []
                            
                            # Define language mappings for better recognition
                            language_mappings = {
                                "en": ["en-US", "en-GB", "en"],
                                "hi": ["hi-IN", "hi", "hi-Latn", "en-IN"],  # Hindi with English fallback
                                "kn": ["kn-IN", "kn", "kn-Latn", "en-IN"]   # Kannada with English fallback
                            }
                            
                            # Get language variants for current language
                            languages = language_mappings.get(self.lang, ["en-US"])
                            
                            # Try multiple recognition engines for better accuracy
                            recognition_engines = [
                                ("Google", self._recognize_google),
                                ("Sphinx", self._recognize_sphinx),
                                ("Azure", self._recognize_azure)
                            ]
                            
                            for engine_name, engine_func in recognition_engines:
                                for lang in languages:
                                    try:
                                        text = engine_func(audio, lang)
                                        if text and len(text.strip()) > 0:
                                            # Voice recognition successful - voice-only interface
                                            return text.lower()
                                    except Exception as e:
                                        pass
                            
                            if not text:
                                # Recognition failed - voice-only interface
                                
                                # Try English as final fallback for non-English languages
                                if self.lang != "en":
                                    try:
                                        text = self.recognizer.recognize_google(audio, language="en-US")
                                        if text:
                                            # English fallback successful - voice-only interface
                                            return text.lower()
                                    except Exception:
                                        pass
                        except Exception as e:
                                        # English fallback failed - voice-only interface
                                        pass
                        except Exception as e:
                            # Recognition error - voice-only interface
                            # Network error - go straight to fallback
                            break
                except Exception as e:
                    # Microphone error - voice-only interface
                    break
                
                retries += 1
                if retries <= max_retries:
                    # Retrying recognition - voice-only interface
                    pass
            
            # Falling back to typed input - voice-only interface
            return self._typed_input_fallback(prompt)
        finally:
            # Restore background music volume after listening attempt
            if hasattr(self, 'audio_manager'):
                self.audio_manager.restore_background_music()

    def _recognize_google(self, audio, language):
        """Google Speech Recognition with enhanced error handling."""
        try:
            return self.recognizer.recognize_google(audio, language=language)
        except (sr.UnknownValueError, sr.RequestError) as e:
            raise Exception(f"Google recognition failed: {e}")
    
    def _recognize_sphinx(self, audio, language):
        """PocketSphinx recognition for offline support."""
        try:
            # PocketSphinx has limited language support, use English for fallback
            if language not in ["en-US", "en-GB", "en"]:
                language = "en-US"
            return self.recognizer.recognize_sphinx(audio, language=language)
        except Exception as e:
            raise Exception(f"Sphinx recognition failed: {e}")
    
    def _recognize_azure(self, audio, language):
        """Azure Speech Recognition (if available)."""
        try:
            # Azure recognition would require API key setup
            # For now, fallback to Google
            return self.recognizer.recognize_google(audio, language=language)
        except Exception as e:
            raise Exception(f"Azure recognition failed: {e}")

    @staticmethod
    def _typed_input_fallback(prompt=None):
        try:
            if prompt:
                reply = input(f"(typed fallback) {prompt}\n> ")
            else:
                reply = input("(typed fallback) Speak or type command:\n> ")
            return reply.strip().lower() if reply else None
        except Exception:
            return None

# -------------------- Intent Parsing --------------------
class IntentRouter:
    def parse_intent(self, text):
        """Parse user intent from text input."""
        if not text:
            return None
        
        # Check for mode requests
        mode = self.detect_mode_request(text)
        if mode:
            return {"type": "mode_change", "mode": mode}
        
        # Check for other intents
        if any(word in text.lower() for word in ["help", "what can you do", "commands"]):
            return {"type": "help"}
        
        if any(word in text.lower() for word in ["stop", "exit", "quit"]):
            return {"type": "stop"}
        
        # Default to general command
        return {"type": "general", "text": text}
    
    def detect_mode_request(self, text):
        if not text:
            return None
        for mode, keys in MODE_KEYWORDS.items():
            for k in keys:
                if k in text:
                    return mode
        return None

    def wants_context_switch(self, text):
        return (text and ("switch to" in text or "change to" in text or "go to" in text))

    def extract_mode_from_switch(self, text):
        if not text:
            return None
        return self.detect_mode_request(text)

    def is_where_am_i(self, text):
        return text and ("where am i" in text or "current location" in text)

    def is_read_command(self, text):
        return text and ("read" in text or "ocr" in text)

    def is_assistant_query(self, text):
        return text and any(k in text for k in ["weather", "reminder", "calendar", "search", "message", "call", "time"])

    def is_sos(self, text):
        return text and any(hw in text for hw in HOTWORDS_SOS)

class Detector:
    def __init__(self, detector_type="yolo", model_path="yolov8n.pt"):
        self.detector_type = detector_type.lower()
        self.model = None
        self.labels = {}
        if self.detector_type == "yolo":
            if YOLO_AVAILABLE:
                try:
                    from ultralytics import YOLO
                    # Convert model_path to string if it's a Path object
                    model_path = str(model_path)
                    self.model = YOLO(model_path)
                except Exception as e:
                    print("[Detector] YOLO model load failed:", e)
                    print("[Detector] Continuing without object detection")
                    self.model = None
            else:
                print("[Detector] ultralytics YOLO not available. Detection disabled.")
        elif self.detector_type == "efficientdet":
            if TF_AVAILABLE:
                try:
                    self.model = hub.load("https://tfhub.dev/tensorflow/efficientdet/lite0/detection/1")
                    labels_url = "https://storage.googleapis.com/openimages/v5/class-descriptions-boxable.csv"
                    labels_path = tf.keras.utils.get_file("class-descriptions-boxable.csv", labels_url)
                    with open(labels_path, "r", encoding="utf-8") as f:
                        for line in f:
                            cid, name = line.strip().split(",")
                            self.labels[cid] = name
                except Exception as e:
                    print("[Detector] EfficientDet load failed:", e)
                    self.model = None
            else:
                print("[Detector] TensorFlow/TensorFlow-Hub not available. Detection disabled.")
        else:
            raise ValueError("Invalid detector type. Use 'yolo' or 'efficientdet'.")

    @staticmethod
    def get_position(x, y, box_w, box_h, frame_w, frame_h):
        """Get detailed position information including distance estimation."""
        # Horizontal position (9 zones for more precision)
        if x < frame_w / 4:
            horizontal = "far left"
        elif x < frame_w / 3:
            horizontal = "left"
        elif x < frame_w * 5/12:
            horizontal = "center left"
        elif x < frame_w * 7/12:
            horizontal = "center"
        elif x < frame_w * 2/3:
            horizontal = "center right"
        elif x < frame_w * 3/4:
            horizontal = "right"
        else:
            horizontal = "far right"
        
        # Vertical position
        if y < frame_h / 3:
            vertical = "high"
        elif y < frame_h * 2/3:
            vertical = "level"
        else:
            vertical = "low"
        
        # Estimate distance based on box size relative to frame
        box_area = box_w * box_h
        frame_area = frame_w * frame_h
        area_ratio = box_area / frame_area
        
        if area_ratio > 0.4:
            distance = "very close"
        elif area_ratio > 0.2:
            distance = "close"
        elif area_ratio > 0.1:
            distance = "nearby"
        elif area_ratio > 0.05:
            distance = "moderate distance"
        elif area_ratio > 0.02:
            distance = "far"
        else:
            distance = "very far"
            
        # Combine position info
        position = f"{distance}, {horizontal}"
        if vertical != "level":
            position += f", {vertical}"
            
        return position, area_ratio

    def detect(self, frame):
        detections = []
        if frame is None:
            return detections
        frame_h, frame_w = frame.shape[:2]
        if self.detector_type == "yolo" and self.model:
            try:
                results = self.model(frame, verbose=False)
                r = results[0]
                # ultralytics result API compatibility
                boxes = getattr(r, "boxes", [])
                for box in boxes:
                    cls_id = int(box.cls[0]) if hasattr(box.cls, "__len__") else int(box.cls)
                    score = float(box.conf[0]) if hasattr(box.conf, "__len__") else float(box.conf)
                    label = self.model.names.get(cls_id, "unknown").lower()
                    xy = box.xyxy[0] if hasattr(box.xyxy, "__len__") else box.xyxy
                    x1, y1, x2, y2 = map(int, xy)
                    box_w, box_h = x2 - x1, y2 - y1
                    cx, cy = (x1 + x2) // 2, (y1 + y2) // 2
                    position, area_ratio = self.get_position(cx, cy, box_w, box_h, frame_w, frame_h)
                    detections.append({
                        "label": label,
                        "pos": position,
                        "box": (x1, y1, x2, y2),
                        "score": score,
                        "area_ratio": area_ratio,
                        "center": (cx, cy)
                    })
            except Exception as e:
                # if detection errors occur, keep going
                print("[Detector] YOLO detect error:", e)
        elif self.detector_type == "efficientdet" and self.model:
            try:
                img_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
                img_tensor = tf.convert_to_tensor(img_rgb, dtype=tf.uint8)[tf.newaxis, ...]
                result = self.model(img_tensor)
                result = {k: v.numpy() for k, v in result.items()}
                for i in range(len(result["detection_scores"])):
                    score = float(result["detection_scores"][i])
                    if score < 0.3:
                        continue
                    class_id = result["detection_classes"][i].decode("utf-8")
                    label = self.labels.get(class_id, "unknown").lower()
                    ymin, xmin, ymax, xmax = result["detection_boxes"][i]
                    x1, y1, x2, y2 = (int(xmin * frame_w), int(ymin * frame_h), int(xmax * frame_w), int(ymax * frame_h))
                    box_w, box_h = x2 - x1, y2 - y1
                    cx, cy = (x1 + x2) // 2, (y1 + y2) // 2
                    position, area_ratio = self.get_position(cx, cy, box_w, box_h, frame_w, frame_h)
                    detections.append({
                        "label": label,
                        "pos": position,
                        "box": (x1, y1, x2, y2),
                        "score": score,
                        "area_ratio": area_ratio,
                        "center": (cx, cy)
                    })
            except Exception as e:
                print("[Detector] EfficientDet detect error:", e)
        return detections

class Onboarding:
    def __init__(self, tts, voice):
        self.tts = tts
        self.voice = voice
        self.steps = [
            "First, make sure your device is charged and positioned comfortably. If it has a camera, keep it clear of any obstruction. Say 'Next' when you're ready.",
            "To talk to me, just say: Hello Assistant. That is my wake word. Say 'Next' to continue.",
            "You can change modes anytime. For example, say: Switch to Walking Mode, or Go to Navigation Mode. Say 'Next' to move on.",
            "You can change the language anytime by saying: Change Language. I will list available languages and you can select one. Say 'Next' to continue.",
            "Remember, I assist you, but you should always keep your cane or guide dog with you outdoors. Say 'Next' to continue.",
            "If you ever feel unsafe, just say: Emergency. I will alert your trusted contact and share your location. Say 'Next' when ready.",
            "If you forget what to say, ask: What can I do? and I'll remind you of available commands. That's it!"
        ]

    def _wait_for_next(self, retries=6):
        """
        Wait until user says 'next' or 'repeat'. If microphone is not responsive,
        fallback to typed input after retries.
        """
        attempts = 0
        while attempts < retries:
            attempts += 1
            cmd = self.voice.listen(timeout=8, phrase_time_limit=4, prompt="Say 'Next' or 'Repeat'.")
            if not cmd:
                # try again
                continue
            if "repeat" in cmd:
                return "repeat"
            if "next" in cmd:
                return "next"
            # accept simple 'n'/'r'
            if cmd.strip() == "n":
                return "next"
            if cmd.strip() == "r":
                return "repeat"
        # fallback: ask typed input explicitly
        fallback = input("No response detected. Type 'next' to proceed, 'repeat' to hear again:\n> ").strip().lower()
        if fallback == "repeat":
            return "repeat"
        return "next"

    def run(self):
        self.tts.speak_blocking(
            "Hello, welcome! I'm your voice assistant, here to help you navigate, walk safely, and manage daily tasks. "
            "I'll give you a short introduction. After each step, say 'Next' to continue, or 'Repeat' to hear it again."
        )

        for i, step in enumerate(self.steps, 1):
            while True:
                self.tts.speak_blocking(f"Step {i}. {step}")
                action = self._wait_for_next()
                if action == "next":
                    break
                # 'repeat' will cause the loop to speak step again

        # Final closure
        self.tts.speak_blocking(
            "You are now ready to start. These are the modes you can choose: Walking Mode detects obstacles around you. "
            "Navigation Mode guides you to your destination. Reading Mode reads text, labels, and money. "
            "Object Mode tells you what's around you. Assistant Mode gives time, weather, and reminders. "
            "Which mode would you like to try first?"
        )

        # Wait until the user speaks a valid mode (or types it)
        mode_choice = None
        attempts = 0
        while not mode_choice and attempts < 6:
            attempts += 1
            cmd = self.voice.listen(timeout=10, phrase_time_limit=6, prompt="Which mode would you like?")
            if not cmd:
                continue
            if "walk" in cmd or "walking" in cmd:
                mode_choice = "walking"
            elif "navigation" in cmd or "navigate" in cmd:
                mode_choice = "navigation"
            elif "read" in cmd or "ocr" in cmd:
                mode_choice = "reading"
            elif "object" in cmd or "scene" in cmd:
                mode_choice = "object_scene"
            elif "assistant" in cmd or "utility" in cmd:
                mode_choice = "assistant"

        if not mode_choice:
            # default fallback
            mode_choice = "walking"
            self.tts.speak_blocking("No selection detected. Defaulting to walking mode.")
        else:
            self.tts.speak_blocking(f"{mode_choice} mode selected. Let's begin.")
        return mode_choice

    def adjust_voice_rate(self, faster=True):
        """Adjust TTS speaking rate."""
        step = 25  # WPM adjustment step
        if faster:
            self.speaking_rate = min(300, self.speaking_rate + step)
            self.tts.set_rate(self.speaking_rate)
            self.tts.speak(f"Speaking rate increased to {self.speaking_rate} words per minute", blocking=True)
        else:
            self.speaking_rate = max(100, self.speaking_rate - step)
            self.tts.set_rate(self.speaking_rate)
            self.tts.speak(f"Speaking rate decreased to {self.speaking_rate} words per minute", blocking=True)
    
    def adjust_volume(self, increase=True):
        """Adjust TTS and sound effect volume."""
        step = 0.1
        if increase:
            self.tts_volume = min(1.0, self.tts_volume + step)
        else:
            self.tts_volume = max(0.1, self.tts_volume - step)
            
        # Update both TTS and sound effect volumes
        self.tts.set_volume(self.tts_volume)
        self.audio_manager.set_volume(self.tts_volume)
        
        # Announce new volume
        percentage = int(self.tts_volume * 100)
        self.tts.speak(f"Volume set to {percentage} percent", blocking=True)

    def guide_obstacles(self):
        now = time.time()
        with self.lock:
            urgent_obstacles = []
            normal_obstacles = []
            
            for obj in self.detected_objects:
                if obj["label"] in OBSTACLES or obj["label"] == "person":
                    # Check if obstacle is too close (based on box size)
                    x1, y1, x2, y2 = obj["box"]
                    box_area = (x2 - x1) * (y2 - y1)
                    frame_area = 640 * 480  # Assumed frame size
                    proximity = box_area / frame_area
                    
                    if proximity > self.urgent_proximity_threshold:
                        urgent_obstacles.append(obj)
                    elif proximity > self.normal_proximity_threshold:
                        normal_obstacles.append(obj)
            
            # Announce obstacles if enough time has passed
            if (now - self.last_obstacle_time) > self.obstacle_cooldown:
                # Urgent obstacles first
                for obj in urgent_obstacles:
                    # Play urgent obstacle sound
                    self.audio_manager.play_obstacle(very_near=True)
                    self.speak(
                        f"Caution! {obj['label']} very close on your {obj['pos']}",
                        message_type="warning",
                        urgent=True,
                        blocking=True
                    )
                
                # Then normal obstacles
                if normal_obstacles:
                    # Play normal obstacle sound once
                    if not urgent_obstacles:  # Only if no urgent sounds played
                        self.audio_manager.play_obstacle(very_near=False)
                    
                    descriptions = []
                    for obj in normal_obstacles[:3]:  # Limit to 3 obstacles
                        descriptions.append(f"{obj['label']} on the {obj['pos']}")
                    
                    if descriptions:
                        self.speak(
                            "Notice: " + ". ".join(descriptions),
                            message_type="info",
                            blocking=False
                        )
                
                self.last_obstacle_time = now


    def check_emergency(self):
        now = time.time()
        with self.lock:
            for obj in self.detected_objects:
                if obj["label"] in EMERGENCIES and (now - self.last_alert_time) > self.alert_cooldown:
                    # Play urgent alert sound
                    self.audio_manager.play_alert(urgent=True)
                    self.speak(f"Emergency detected: {obj['label']} at {obj['pos']}", blocking=True)
                    self.last_alert_time = now
                    self.emergency_assistance(reason=f"Detected {obj['label']}")

    def object_scene_tick(self):

    def assistant_tick(self, last_command=None):

        with self.lock:

    # Detection loop
    def detection_loop(self):
        """
        Optimized detection loop with better error handling and performance
        """
        if not self._cap:
            print("[Camera] No camera initialized")
            return
            
        try:
            frame_skip = 3  # Process every nth frame for better performance
            frame_count = 0
            last_process_time = time.time()
            fps_target = 30
            frame_time = 1.0 / fps_target
            
            while not self._stop_event.is_set():
                # Frame capture with timeout
                try:
                    ret, frame = self._cap.read()
                    if not ret:
                        time.sleep(0.05)
                        continue
                except Exception as e:
                    print(f"Camera error: {e}")
                    time.sleep(0.1)
                    continue

                frame_count += 1
                current_time = time.time()
                
                # Skip frames for performance
                if frame_count % frame_skip != 0:
                    # Still show camera feed for developer
                    cv2.imshow("Developer Camera View", frame)
                    if cv2.waitKey(1) & 0xFF == ord('q'):
                        self._stop_event.set()
                        break
                    continue

                # Only run detection if we have a detector
                filtered = []
                if self.detector and self.detector.model:
                    try:
                        detections = self.detector.detect(frame)
                        for d in detections:
                            lbl = d["label"]
                            if (lbl in STREET_CLASSES) or (lbl in OBSTACLES) or (lbl in EMERGENCIES):
                                filtered.append(d)

                                # draw boxes for developer
                                x1, y1, x2, y2 = d["box"]
                                cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
                                cv2.putText(frame, f"{lbl} {d['score']:.2f}",
                                          (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX,
                                          0.6, (0, 255, 0), 2)
                    except Exception as e:
                        print(f"Detection error: {e}")

                # Update detected objects thread-safely
                with self.lock:
                    self.detected_objects = filtered

                # Run mode-specific logic based on current mode
                try:
                    self.check_emergency()
                    if self.current_mode == "walking":
                        self.walking_tick()
                    elif self.current_mode == "navigation":
                        self.navigation_tick()
                    elif self.current_mode == "reading":
                        self.reading_tick(frame)
                    elif self.current_mode == "object_scene":
                        self.object_scene_tick()
                except Exception as e:
                    print(f"Mode processing error: {e}")

                # Show developer camera window
                cv2.imshow("Developer Camera View", frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    self._stop_event.set()
                    break

                # Maintain consistent frame rate
                elapsed = time.time() - current_time
                if elapsed < frame_time:
                    time.sleep(frame_time - elapsed)
        finally:
            self._close_camera()
            cv2.destroyAllWindows()

    def command_loop(self):
        """Enhanced command loop with context awareness and interrupt handling."""
        while not self._stop_event.is_set():
            try:
                # Listen for command with enhanced VoiceIO
                cmd = self.voice.listen(
                    timeout=6,
                    phrase_time_limit=6,
                    prompt="Say a command."
                )
                
                if not cmd:
                    continue

                # Parse intent with enhanced router
                intent = self.router.parse_intent(cmd)
                
                # Add to conversation history
                self.state_manager.add_to_history("user", cmd)
                
                # Handle intent based on type
                if intent.type == IntentType.EMERGENCY:
                    self.voice.speak("Emergency detected!", ResponseType.ERROR)
                    self.emergency_assistance(reason=cmd)
                    continue
                
                if intent.type == IntentType.MODE_CHANGE:
                    mode = intent.data.get("mode")
                    if mode:
                        # Check if we need to handle interrupts
                        if self.router.should_interrupt(cmd):
                            current_tasks = self.state_manager.get_active_tasks()
                            if current_tasks:
                                self.voice.speak(
                                    "I notice you're in the middle of something. "
                                    "Should I pause that first?",
                                    ResponseType.CLARIFICATION
                                )
                                response = self.voice.listen(timeout=5)
                                if response and any(word in response.lower() 
                                                  for word in ["yes", "yeah", "sure"]):
                                    for task in current_tasks:
                                        self.task_manager.pause_task(task.name)
                        
                        self.change_mode(mode)
                    continue
                
                if intent.type == IntentType.CONTROL:
                    action = intent.data.get("action")
                    if action == "pause":
                        self._handle_pause_request()
                    elif action == "resume":
                        self._handle_resume_request()
                    elif action == "stop":
                        self._handle_stop_request()
                    continue
                
                if intent.type == IntentType.EMOTIONAL:
                    self._handle_emotional_response(intent.data.get("emotion"))
                    continue
                
                # Handle other intents
                if self.router.is_where_am_i(cmd):
                    self.voice.speak(
                        "Location services not connected. Please enable GPS.",
                        ResponseType.ERROR
                    )
                    continue
                
                if self.router.is_read_command(cmd):
                    if self.current_mode != "reading":
                        self.change_mode("reading")
                    self.voice.speak(
                        "Reading mode ready. Hold text steady.",
                        ResponseType.CONFIRMATION
                    )
                    continue
                
                # Language change commands
                if any(phrase in cmd for phrase in ["change language", "switch language", "language settings", "change to"]):
                    self.change_language_interactive()
                    continue
                
                # OCR commands
                if any(phrase in cmd for phrase in ["check ocr", "ocr status", "text recognition"]):
                    self.check_ocr_status()
                    continue
                
                if any(phrase in cmd for phrase in ["read text", "read", "ocr"]):
                    if self.current_mode != "reading":
                        self.change_mode("reading")
                    self.speak("Reading mode activated. Point camera at text to read.", blocking=True)
                    continue
                
                # Voice control commands
                if cmd in self.voice_commands:
                    self.voice_commands[cmd]()
                    continue
                
                # Assistant queries with context
                if self.router.is_assistant_query(cmd):
                    if self.current_mode != "assistant":
                        self.change_mode("assistant")
                    
                    # Get recent context
                    context = self.state_manager.get_recent_context()
                    self.assistant_tick(
                        last_command=cmd,
                        context=context
                    )
                    continue
                
                # Object queries with enhanced response
                with self.lock:
                    hits = [o for o in self.detected_objects 
                           if any(w in cmd for w in o["label"].split())]
                if hits:
                    msg = ". ".join([f"{o['label']} at {o['pos']}" 
                                   for o in hits[:3]])
                    self.voice.speak(msg, ResponseType.CONFIRMATION)
                else:
                    self.voice.speak(
                        "I don't see that yet. Would you like me to "
                        "notify you when I detect it?",
                        ResponseType.CLARIFICATION
                    )
                    
            except Exception as e:
                print(f"Command loop error: {e}")
                self.voice.speak(
                    "I encountered an error. Please try again.",
                    ResponseType.ERROR
                )

    # Boot flow including language selection and onboarding
    def _select_language(self):
        """Handle the language selection process by listening for user's choice."""
        # For now, assume all languages are available
        # In the future, this can check for actual model availability
        available_models = {"en", "hi", "kn"}

        # Play welcome sound
        if self.audio_manager:
            self.audio_manager.play('tutorial_start')
            time.sleep(1)
        
        # Initialize TTS in English
        if self.tts:
            self.tts.set_lang("en")
            time.sleep(0.5)
            self.speak("Welcome to BlindNav Plus Plus!", blocking=True)
            time.sleep(0.5)
            self.speak("I will help you select your preferred language.", blocking=True)
            time.sleep(0.5)
        
        # Define available languages and their keywords
        languages_info = {
            "en": {
                "name": "English", 
                "keywords": ['english', 'inglish', 'ing', 'en', 'one', 'first', 'english language', 'english mode']
            },
            "hi": {
                "name": "Hindi", 
                "keywords": ['hindi', 'हिंदी', 'हिन्दी', 'hi', 'two', 'second', 'hindi language', 'hindi mode', 'hindee', 'hinde']
            },
            "kn": {
                "name": "Kannada", 
                "keywords": ['kannada', 'ಕನ್ನಡ', 'kannad', 'kn', 'three', 'third', 'kannada language', 'kannada mode', 'kannad', 'kannad language']
            }
        }
        
        # List available languages
        available_languages = []
        for code, info in languages_info.items():
            if code in available_models:
                available_languages.append((code, info))
        
        if not available_languages:
            self.speak("No language models found. Using English as default.", blocking=True)
            return "en"
        
        # Announce available languages
        self.speak("Available languages are:", blocking=True)
        for i, (code, info) in enumerate(available_languages, 1):
            self.speak(f"{i}. {info['name']}", blocking=True)
            time.sleep(0.3)
        
        self.speak("Please say the name of your preferred language, or say the number.", blocking=True)
        time.sleep(0.5)
        
        max_attempts = 3
        for attempt in range(max_attempts):
            if self.audio_manager:
                self.audio_manager.play('ready')
                time.sleep(0.5)
            
            # Try to listen for voice input
            response = self.voice.listen(timeout=10)
            if response:
                response = response.lower().strip()
                
                # Check for number selection
                if response in ['1', 'one', 'first']:
                    return available_languages[0][0]
                elif response in ['2', 'two', 'second'] and len(available_languages) > 1:
                    return available_languages[1][0]
                elif response in ['3', 'three', 'third'] and len(available_languages) > 2:
                    return available_languages[2][0]
                
                # Check for language name selection
                for code, info in available_languages:
                    if any(word in response for word in info["keywords"]):
                        return code
                
                # If no valid language detected
                if attempt < max_attempts - 1:
                    self.speak("I didn't understand your selection. Please say the language name or number.", blocking=True)
                    time.sleep(0.5)
            else:
                # No response - try fallback to typed input
                if attempt < max_attempts - 1:
                    self.speak("I didn't hear anything. Please try again.", blocking=True)
                    time.sleep(0.5)
                    # Fallback to typed input for testing
                    try:
                        fallback_input = input("(Fallback) Type language choice (1/2/3 or en/hi/kn): ").strip().lower()
                        if fallback_input in ['1', 'one']:
                            return available_languages[0][0]
                        elif fallback_input in ['2', 'two'] and len(available_languages) > 1:
                            return available_languages[1][0]
                        elif fallback_input in ['3', 'three'] and len(available_languages) > 2:
                            return available_languages[2][0]
                        elif fallback_input in ['en', 'hi', 'kn'] and fallback_input in available_models:
                            return fallback_input
                    except:
                        pass
        
        # Default to first available language
        self.speak("No valid selection detected. Using default language.", blocking=True)
        return available_languages[0][0]

    def run(self, camera_index=0):
        """Main run loop with complete voice-driven workflow."""
        try:
            # Start loading music (play once only, not looping)
            loading_music = MusicPlayer("loading_music.mp3")
            loading_music.play_once()
            loading_music.dim()  # Start with dimmed music
            time.sleep(1)
            
            # Stop music before starting voice workflow
            loading_music.stop()
            time.sleep(1)

            # Complete voice-driven startup workflow
            self._voice_driven_startup()
            
            # After voice-driven startup is complete, start camera
            try:
                self._open_camera(camera_index=camera_index)
            except Exception as e:
                self.speak("Camera not available. Please connect the camera.", blocking=True)
                self.speak("Camera error occurred. Continuing without camera.", blocking=True)
                return

            # Start detection and command threads only after setup is complete
            det_thread = threading.Thread(target=self.detection_loop, daemon=True)
            cmd_thread = threading.Thread(target=self.command_loop, daemon=True)
            det_thread.start()
            cmd_thread.start()

            # main loop
            try:
                while det_thread.is_alive() and cmd_thread.is_alive():
                    time.sleep(0.5)
            except KeyboardInterrupt:
                print("Interrupted by user.")
        finally:
            self._stop_event.set()
            self._close_camera()
            
            # Play shutdown sound
            if hasattr(self, 'audio_manager'):
                self.audio_manager.play_shutdown()
                time.sleep(0.5)  # Give time for shutdown sound
                
            # Clean up audio systems
            if hasattr(self, 'tts'):
                self.tts.stop()
                
            print("Navigator stopped.")

    def _safe_update_ui(self, func):
        """Safely execute UI updates with proper error handling."""
        if not self.root or not self.root.winfo_exists():
            return
        try:
            # Use after_idle to ensure UI updates happen in the main thread
            self.root.after_idle(lambda: self._execute_ui_update(func))
        except Exception as e:
            print(f"UI Update Error: {e}")
            
    def _execute_ui_update(self, func):
        """Execute the UI update function with error handling."""
        try:
            func()
            self.root.update_idletasks()
        except Exception as e:
            print(f"UI Update Execution Error: {e}")

        # Prevent multiple starts
        if self.system_running or self.navigator:

                self.start_button.configure(state="disabled"),
                self.stop_button.configure(state="disabled"),
                self.status_label.configure(text="System Status: STARTING...")

                self.status_label.configure(text="System Status: RUNNING"),
                self.stop_button.configure(state="normal")

                self.navigator._stop_event.set()

    # Direct voice-driven startup (bypass Tkinter control panel)
